\begin{problem}
Consider a growth model with preferences $\sum_{t=0}^{\infty} \beta^{t} \log C_{t},$ production function $Y_{t}=A K_{t}^{\alpha}$, the capital law of motion $K_{t+1}=K_{t}^{1-\delta} I_{t}^{\delta},$ and the resource constraint $Y_{t}=C_{t}+I_{t}$
\begin{subproblem}
    Write down the social planner's problem and derive the Euler equation. Provide the intuition to this optimality condition using the perturbation argument.
\end{subproblem}
\begin{proof}[Answer]
 The social planner solves the following problem:
\begin{maxi*}|s|
{C_t, I_t, K_{t+1}}{\sum_{t=0}^\infty \beta^t \log C_t}
{}{}
\addConstraint{C_t + I_t = AK_t^\alpha}{}
\addConstraint{K_{t+1}=K_t^{1-\delta} I^{\delta}}{}
\end{maxi*}
\end{proof}

\textbf{Note:} This problem have a superfluous constraint and optimization variable, since $K_{t+1}$ and $I_t$ are determined jointly given $K_t$. I took this approach since I found that the algebra was easier.

Since the objective function is concave, we can use the Lagrangian to solve:

$$\mathcal{L}(\cdot) = \sum_{t=0}^\infty \beta^t \log C_t - \lambda_t\left(C_t + I_t - AK_t^\alpha \right) - \mu_t\left(K_{t+1} - K_t^{1-\delta} I^{\delta} \right)$$


Taking first order conditions we get:

\begin{align*}
    C_t:&\qquad  \frac{\beta^t}{C_t} = \lambda_t \\
    I_t:&\qquad \delta\mu_t\left( \frac{K_t}{I_t}\right)^{1-\delta}=\lambda_t\\
    K_{t+1}:&\qquad \mu_t = \alpha A \lambda_{t+1}K_{t+1}^{\alpha-1}+\mu_{t+1}(1-\delta)\left( \frac{I_t}{K_t}\right)^{\delta}
\end{align*}

Combining the $C_t$ and  $I_t$ FOC's we get:

$$\mu_t = \frac{\beta^t}{\delta C_t} \left( \frac{I_t}{K_t}\right)^{1-\delta}$$

And plugging the value of $\mu_t$ and $\lambda_t$ in the FOC associated with $K_{t+1}$:

$$\frac{\beta^t}{\delta C_t} \left( \frac{I_t}{K_t}\right)^{1-\delta} = \alpha A \frac{\beta^{t+1}}{C_{t+1}}K_t^{\alpha-1}+\frac{\beta^{t+1}}{\delta C_{t+1}} \left( \frac{I_{t+1}}{K_{t+1}}\right)^{1-\delta}(1-\delta)\left( \frac{I_t}{K_t}\right)^{\delta}

Simplifying the above expression:

\begin{equation}\label{euler}\tag{EE}
    \frac{C_{t+1}}{C_t} = \delta\beta\left[ 
    \alpha A K_{t+1}^{\alpha-1}+\frac{(1-\delta)}{\delta}\left( \frac{I_{t+1}}{K_{t+1}} \right)
    \right]
\end{equation}

Note that:

\begin{align*}
    K_{t+2} = K_{t+1}^\delta I_{t+1}^{1-\delta}\qiq & \frac{\partial K_{t+2}}{\partial I_{t+1}} = \delta K_{t+1}^{1-\delta}I_{t+1}^{\delta-1} \\
    & \frac{\partial K_{t+2}}{\partial K_{t+1}} = (1-\delta) K_{t+1}^{-\delta}I_{t+1}^{\delta}
\end{align*}

Therefore we can write the Euler equation as:

\begin{equation}\label{euler}\tag{EE'}
    \frac{U'(C_t)}{U'(C_{t+1})} = \delta\beta\left[ 
    F'(K_{t+1})+\frac{\partial K_{t+2}/\partial K_{t+1}}{\partial K_{t+2}/\partial I_{t+1}} 
    \right]
\end{equation}

To find steady state we set:

$$C_t = C_{t+1} = \overline{C} \qquad K_t = K_{t+1} = \overline{K} \qquad I_t = I_{t+1} = \overline{I}$

After some algebra we get:

\begin{align*}
    \overline{K}&=\left( \frac{\alpha \beta \delta A}{1-\beta(1-\delta)} \right)^{\frac{1}{1-\alpha}}\\
    \overline{I}&=\overline{K}\\
    \overline{C}&=A\overline{K}^\alpha - \overline{K}
\end{align*}
\end{proof}
